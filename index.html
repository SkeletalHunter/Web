    using System;
using Battles;
using System.Collections.Generic;
using Lab2;

namespace Actions
{
    class Actions
    {
        public static void DealDamage
        (BattleUnitsStack Stack,
        int Damage)
        {
            int HitPoints = (Stack.GetCountInfo() - 1)
            * Stack.GetUnitInfo().GetHitPointsInfo()
            + Stack.GetCurHitPoints();
            if (HitPoints > Damage)
            {
                int NewHitPoints = (HitPoints - Damage)
                % Stack.GetUnitInfo().GetHitPointsInfo();
                if (NewHitPoints == 0)
                {
                    Stack.StatsSetHitPoints(Stack.GetUnitInfo()
                    .GetHitPointsInfo());
                    Stack.StatsSetCount((HitPoints - Damage)
                    / Stack.GetUnitInfo().GetHitPointsInfo());
                }
                else
                {
                    Stack.StatsSetHitPoints(NewHitPoints);
                    Stack.StatsSetCount((HitPoints - Damage)
                    / Stack.GetUnitInfo().GetHitPointsInfo() + 1);
                }
            }
            else
            {
                Stack.StatsSetHitPoints(0);
                Stack.StatsChangeCount(0);
            }
        }

        public static void EndOfUnitTurn(BattleUnitsStack Unit)
        {

        }

        public static BattleUnitsStack ChooseUnit(BattleArmy Army)
        {
            Console.WriteLine("Choose the stack to attack");
            int count = 0;
            string input;
            foreach (BattleUnitsStack ArmyUnit in Army.GetSquad())
            {
                count++;
                Console.WriteLine(count
                + ": "
                + ArmyUnit.GetType()
                + "["
                + ArmyUnit.GetCurCount()
                + "]");
            }
            while (true)
            {
                input = Console.ReadLine();
                if (int.TryParse(input, out int x))
                {
                    if (x > count
                    || Army.GetSquad()[x - 1].GetCurCount()
                    == 0)
                    {
                        if (x > count)
                        {
                            Console.WriteLine
                            ("That stack does not exist");
                        }
                        else
                        {
                            Console.WriteLine
                            ("That stack is dead");
                        }
                    }
                    else
                    {
                        return Army.GetSquad()[x - 1];
                    }
                }
                else
                {
                    Console.WriteLine
                    ("That is not a number");
                }
            }
        }
    }
